    private final Properties config = new Properties();

    public SimpleContainer() {

    public void register(Class<?> configClass) {
        try {
            Configuration configuration = configClass.getAnnotation(Configuration.class);
            if (configuration == null) {
                throw new IllegalArgumentException("é¡åˆ¥å¿…é ˆæœ‰ @Configuration è¨»è§£");
            }

            Object configInstance = configClass.getDeclaredConstructor().newInstance();

            // è™•ç† @Bean æ–¹æ³•

                if (bean != null) {

                        method.getName() : bean.name();

                    beanMethods.put(beanName, method);

                        Object beanInstance = method.invoke(configInstance);

        } catch (Exception e) {

        }
    }

            // å»¶é²å‰µå»º

                field.setAccessible(true);

                if (value != null) {

        if (type == int.class || type == Integer.class) return Integer.parseInt(value);
        if (type == boolean.class || type == Boolean.class) return Boolean.parseBoolean(value);
        return value;
    }

class SimpleWebFramework {
    private final Map<String, Object> controllers = new HashMap<>();

    public void registerController(Object controller) {
        Class<?> clazz = controller.getClass();
        Controller controllerAnnotation = clazz.getAnnotation(Controller.class);

        if (controllerAnnotation != null) {
            String basePath = controllerAnnotation.path();
            controllers.put(basePath, controller);
            System.out.println("è¨»å†Šæ§åˆ¶å™¨: " + clazz.getName() + " -> " + basePath);
        }
    }

    public String handleRequest(String path, RequestMethod method, Map<String, String> params) {
        for (Map.Entry<String, Object> entry : controllers.entrySet()) {
            String basePath = entry.getKey();
            Object controller = entry.getValue();

            if (path.startsWith(basePath)) {
                return processController(controller, path.substring(basePath.length()), method, params);
            }
        }
        return "404 Not Found";
    }

    private String processController(Object controller, String path, RequestMethod method, Map<String, String> params) {
        Method[] methods = controller.getClass().getDeclaredMethods();

        for (Method methodObj : methods) {
            RequestMapping mapping = methodObj.getAnnotation(RequestMapping.class);
            if (mapping != null && mapping.method() == method) {
                if (matchPath(mapping.value(), path)) {
                    try {
                        return invokeMethod(controller, methodObj, path, params);
                    } catch (Exception e) {
                        return "500 Internal Server Error: " + e.getMessage();
                    }
                }
            }
        }
        return "404 Not Found";
    }

    private boolean matchPath(String pattern, String path) {
        // ç°¡åŒ–çš„è·¯å¾‘åŒ¹é…
        if (pattern.equals(path)) return true;
        if (pattern.contains("{") && pattern.contains("}")) {
            // ç°¡å–®çš„åƒæ•¸è·¯å¾‘åŒ¹é…
            String regex = pattern.replaceAll("\\{[^}]+\\}", "[^/]+");
            return path.matches(regex);
        }
        return false;
    }

    private String invokeMethod(Object controller, Method method, String path, Map<String, String> params) throws Exception {
        Parameter[] parameters = method.getParameters();
        Object[] args = new Object[parameters.length];

        for (int i = 0; i < parameters.length; i++) {
            Parameter param = parameters[i];

            // è™•ç† @RequestParam
            RequestParam requestParam = param.getAnnotation(RequestParam.class);
            if (requestParam != null) {
                String paramName = requestParam.value();
                String value = params.get(paramName);
                if (value == null && requestParam.required() && !requestParam.defaultValue().isEmpty()) {
                    value = requestParam.defaultValue();
                }
                args[i] = convertParameter(value, param.getType());
                continue;
            }

            // è™•ç† @PathVariable
            PathVariable pathVariable = param.getAnnotation(PathVariable.class);
            if (pathVariable != null) {
                String value = extractPathVariable(path, pathVariable.value());
                args[i] = convertParameter(value, param.getType());
                continue;
            }
        }

        return (String) method.invoke(controller, args);
    }

    private Object convertParameter(String value, Class<?> type) {

    public static ValidationResult validate(Object obj) {
        ValidationResult result = new ValidationResult();
        Class<?> clazz = obj.getClass();

        Field[] fields = clazz.getDeclaredFields();
        for (Field field : fields) {

                result.addError("ç„¡æ³•å­˜å–æ¬„ä½: " + field.getName());

    private static void validateField(Field field, Object value, ValidationResult result) {
        // @NotNull é©—è­‰

        if (notNull != null && value == null) {
